12) src/lib/cart.ts (локальная корзина через cookies)

'use server'
import { cookies } from 'next/headers'
import { z } from 'zod'

const CartSchema = z.array(z.object({
productId: z.string(),
sizeId: z.string().optional(),
qty: z.number().int().min(1)
}))

const CART_KEY = 'ecocraft_cart'

export async function getCart() {
const raw = (await cookies()).get(CART_KEY)?.value
try { return CartSchema.parse(JSON.parse(raw ?? '[]')) } catch { return [] }
}

export async function addToCart(item: { productId: string, sizeId?: string, qty?: number }) {
const cart = await getCart()
const idx = cart.findIndex(i => i.productId === item.productId && i.sizeId === item.sizeId)
if (idx >= 0) cart[idx].qty += item.qty ?? 1
else cart.push({ productId: item.productId, sizeId: item.sizeId, qty: item.qty ?? 1 })
;(await cookies()).set(CART_KEY, JSON.stringify(cart), { httpOnly: true, path: '/' })
}

export async function removeFromCart(productId: string, sizeId?: string) {
const cart = (await getCart()).filter(i => !(i.productId === productId && i.sizeId === sizeId))
;(await cookies()).set(CART_KEY, JSON.stringify(cart), { httpOnly: true, path: '/' })
}

export async function clearCart() {
;(await cookies()).set(CART_KEY, '[]', { httpOnly: true, path: '/' })
}

function mapParam(v: string | string[] | null) {
return typeof v === 'string' && v.length ? v : undefined
}

export default async function CatalogPage({ searchParams }: { searchParams: Record<string, string | string[] | undefined> }) {
const where: any = {}
const category = mapParam(searchParams.category ?? null)
const material = mapParam(searchParams.material ?? null)
const warmth = mapParam(searchParams.warmth ?? null)
if (category) where.category = category
if (material) where.material = material
if (warmth) where.warmth = warmth

const products = await prisma.product.findMany({ where, orderBy: { createdAt: 'desc' }, include: { images: true } })

return (
<div className="space-y-6">
<h1 className="text-2xl font-medium">Каталог</h1>
<Filters />
{products.length === 0 && (

## 28) Заметки по дальнейшей интеграции платежей

- На этапе подтверждения заказа сохранять черновик в БД; при успешном колбэке платежа менять статус на `PAID`.
- Для ЮKassa/Робокассы — отдельные API‑роуты для `createPayment` и `webhook`.
